void gyro()

{
          int x = analogRead(A0);
          int y = analogRead(A1);
          int z = analogRead(A2);

          x = x / 100;
          y = y / 100;
          z = z / 100;
          /******************/
//Serial.print(x);
//Serial.print(y);
//Serial.print(z);
//delay(2000);

      if(x==3&&y==3&&z==3)
      {
    //   g=0;
digitalWrite(Buzzer,LOW); 
lcd.setCursor(14,0);
lcd.print("FN");
Serial.print('n');            
      }

      else if(x==3&&y==3&&z==4)
      {       
  //    g=0;
digitalWrite(Buzzer,LOW); 
lcd.setCursor(14,0);
lcd.print("FN");
Serial.print('n'); 
delay(5);    
      }

      else
      {

   //   g=1;
digitalWrite(Buzzer,HIGH);
lcd.setCursor(14,0);
lcd.print("FD");
Serial.print('N');
delay(5);
//            gps(help); 


      }
}
